import discord
from discord.ext import commands
import asyncio
import youtube_dl
import pafy
from discord import Spotify
import DiscordUtils

music = DiscordUtils.Music()

class music(commands.Cog):
	def __init__(self, client):
		self.client = client
		self.song_queue = {}

		self.setup()
	def setup(self):
		for guild in self.client.guilds:
			self.song_queue[guild.id] = []

	async def check_queue(self, ctx):
		if len(self.song_queue[ctx.guild.id]) > 0:
			await self.play_song(ctx, self.song_queue[ctx.guild.id][0])
			self.song_queue[ctx.guild.id].pop(0)

	async def search_song(self, amount,song, get_url = False):
		info = await self.client.loop.run_in_executor(None, lambda: youtube_dl.YoutubeDL({'format': 'bestaudio', 'quiet': True}).extract_info(f'ytsearch{amount}:{song}', download = False, ie_key = 'YoutubeSearch'))
		if len(info['entries']) == 0: return None

		return [entry['webpage_url'] for entry in info['entries']] if get_url else info

	async def play_song(self, ctx, song):
		url = pafy.new(song).getbestaudio().url
		ctx.voice_client.play(discord.PCMVolumeTransformer(discord.FFmpegPCMAudio(url)), after = lambda error: self.client.loop.create_task(self.check_queue(ctx)))
		ctx.voice_client.source_volume = 0.5


	@commands.command()
	async def join(self, ctx):
		if ctx.author.voice is None:
			return await ctx.send(f'{ctx.author} you are not currently in any voice channel')
		if ctx.voice_client is True:
			return await ctx.send('I am already in a voice channel')

		await ctx.author.voice.channel.connect()
		await ctx.send(f'Joined {ctx.message.author.voice.channel.name}')

	@commands.command(aliases = ['dc','leave'])
	async def disconnect(self, ctx):
		if ctx.voice_client is not None:
			return await ctx.voice_client.disconnect()

		await ctx.send(f'{ctx.author} I am not currently in any voice channel')

	@commands.command()
	async def play(self, ctx, *, song=None):
		if song is None:
			return await ctx.send(f'You need to mention a song to play, {ctx.author}')
		if ctx.voice_client is None:
			return await ctx.send('I am not currently in any voice channel to play this song')
		if not ('youtube.com/watch?=' in song or 'https://youtu.be/' in song):
			await ctx.send('Searching for song, please wait')

			result = await self.search_song(1, song, get_url = True)

			if result is None:
				return await ctx.send('I could not find the given song, try using my search command instead')
			song = result[0]

		if ctx.voice_client.source is not None:
			queue_len = len(self.song_queue[ctx.guild.id])

			if queue_len < 101:
				self.song_queue[ctx.guild.id].append(song)
				return await ctx.send(f'Song added to queue position {queue_len+1}')

			else:
				return await ctx.send('I can only queue up to 100 songs, please wait for this song to finish to add to the queue')

		await self.play_song(ctx, song)
		await ctx.send(f'Now playing: {song}')

	@commands.command()
	async def search(self, ctx, *, song = None):
		if song is None: return await ctx.send(f'You didn\'t include any song to search for, {ctx.author.mention}')

		await ctx.send('Searching for song, please wait...')

		info = await self.search_song(5, song)

		embed = discord.Embed(title = f'Showing results for {song}:', description = 'Use these URLs to play the song that you want, if the one I displayed was incorrect\n')

		amount = 0 

		for entry in info['entries']:
			embed.description += f"[{entry['title']}]({entry['webpage_url']})\n"
			amount += 1

		embed.set_footer(text = f'Displaying the first {amount} results')
		await ctx.send(embed = embed)

	@commands.command(aliases = ['q'])
	async def queue(self, ctx):
		if len(self.song_queue[ctx.guild.id]) == 0:
			return await ctx.send('There are currently no songs in the queue')

		embed = discord.Embed(title = 'Song Queue:', description = '', color = 0x008EFD)
		i = 1

		for url in self.song_queue[ctx.guild.id]:
			embed.description += f'{i}.) {url}\n'

			i+= 1

		embed.set_footer(text = f'Next song queued: {url} ')
		await ctx.send(embed = embed)

	@commands.command()
	async def skip(self, ctx):
		if ctx.voice_client is None:
			return await ctx.send('I am currently not playing any songs')
		if ctx.author.voice is None:
			return await ctx.send('You are currently not in any voice channels')

		if ctx.author.voice.channel.id != ctx.voice_client.channel.id:
			return await ctx.send('I am currently not in your voice channel')

		poll = discord.Embed(title = f'Vote to skip by: {ctx.author.name}{ctx.author.discriminator}', description = '**80% of the vc must agree to skip this song', color = 0x00FDF9)
		poll.add_field(name = 'Skip', value = ':white_check_mark:')
		poll.add_field(name = 'Don\'t Skip', value = ':no_entry:sign:')
		poll.set_footer(text = 'Voting ends in 15 seconds')

		poll_msg = await ctx.send(embed = poll)
		poll_id = poll_msg.id

		await poll_msg.add_reaction(u'\u2705')
		await poll_msg.add_reaction(u'\U0001F6AB')

		await asyncio.sleep(15)

		poll_msg = await ctx.channel.fetch_message(poll_id)

		votes = {u'\u2705': 0, u'\U0001F6AB': 0}
		reacted = []

		for reaction in poll_msg.reactions:
			if reaction.emoji in [u'\u2705', u'\U0001F6AB']:
				async for user in reaction.users():
					if user.voice.channel.id == ctx.voice_client.channel.id and user.id not in reacted and not user.bot:
						votes[reaction.emoji] += 1

						reacted.append(user.id)


		skip = False

		if votes[u'\u2705'] > 0:
			if votes[u'\U0001F6AB'] == 0 or votes[u'\u2705']/(votes[u'\u2705'] + votes[u'\U0001F6AB']) > 0.79:
				skip = True
				embed = discord.Embed(title = 'Skip successful!', description = 'Voting to skip the song was sucessful', color = 0xFDC000)

		if not skip:
			embed = discord.Embed(title = 'Skip failed', description = 'Voting to skip the song was in majority not to skip', color = 0xFD6B00)

		embed.set_footer(text = 'Voting ended at: {}'.format(strftime("%I:%M %p UTC")))

		await poll_msg.clear_reactions()
		await poll_msg.edit(embed = embed)

		if skip:
			ctx.voice_client.stop()

		@commands.command()
		async def pause(self, ctx):
			if ctx.voice_client is None:
				await ctx.send('I am not in any voice channel')
			if ctx.voice_client.is_paused():
				return await ctx.send('I am already paused')

			ctx.voice_client.pause()
			await ctx.send('I have paused the current song')

		@commands.command()
		async def resume(self, ctx):
			if ctx.voice_client is None:
				await ctx.send('I am not in any voice channel')
			if not ctx.voice_client.is_paused():
				await ctx.send('I am already playing a song')

			ctx.voice_client.resume()
			await ctx.send('Resuming the current song')




		


def setup(client):
	client.add_cog(music(client))
